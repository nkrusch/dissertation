% ================================ %
% SOURCE CODE HIGHLIGHTING OPTIONS
% ================================ %

% plain C code - without colors
\lstdefinestyle{PlainStyle}{
    basicstyle=\small\ttfamily,
    breakatwhitespace=false,
    breaklines=true,
    captionpos=b,
    keepspaces=true,
    showspaces=false,
    escapechar=!,
    escapeinside={(*}{*)},
    showstringspaces=false,
    showtabs=false,
    tabsize=2,
    language=C
}

% C code highlight
\lstdefinestyle{CStyle}{
    basicstyle=\small\ttfamily,
    commentstyle=\color{mGreen},
    keywordstyle=\color{mPurple}\bfseries,
    stringstyle=\color{magenta},
    breakatwhitespace=false,
    breaklines=true,
    captionpos=b,
    keepspaces=true,
    escapechar=!,
    showspaces=false,
    showstringspaces=false,
    showtabs=false,
    tabsize=2,
    language=C
}
