% programming languages
\newcommand{\pr}{\lstinline[mathescape,language=imp,basicstyle=\ttfamily,escapeinside={(*}{*)}]}
\newcommand{\prc}{\lstinline[language=C,basicstyle=\ttfamily\bfseries]}
\lstdefinelanguage{imp}{morekeywords={if,then,else,while,do,loop}}

% plain C code - without colors
\lstdefinestyle{PlainStyle}{
    basicstyle=\small\ttfamily,
    breakatwhitespace=false,
    breaklines=true,
    captionpos=b,
    keepspaces=true,
    showspaces=false,
    escapechar=!,
    escapeinside={(*}{*)},
    showstringspaces=false,
    showtabs=false,
    tabsize=2,
    language=C
}
